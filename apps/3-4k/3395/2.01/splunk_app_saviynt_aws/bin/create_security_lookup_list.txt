user_account.csv::::select distinct USERNAME,SYSTEMUSERNAME,TITLE,FIRSTNAME,MIDDLENAME,LASTNAME,COUNTRY,CITY,STATE,STREET,NAME as ACCOUNT_NAME,SYSTEMNAME,ENDPOINTNAME,DATE(created_on) as created_on,DATE(lastlogondate) as logon_date from users,user_accounts,accounts,endpoints,securitysystems where users.userkey = user_accounts.userkey and user_accounts.accountkey = accounts.accountkey and accounts.endpointkey = endpoints.endpointkey and endpoints.securitysystemkey = securitysystems.systemkey and accounts.status=1
ac_ent.csv::::select distinct CAST(ac.name AS Char(32)) as NAME , CAST(ev.entitlement_value AS Char(32)) as ENTITLEMENT_VALUE,ENTITLEMENTNAME,ENDPOINTNAME from accounts ac ,account_entitlements1 ae,entitlement_values ev,entitlement_types et ,endpoints ep where ac.accountkey=ae.accountkey and ae.entitlement_valuekey=ev.entitlement_valuekey and ev.entitlementtypekey= et.entitlementtypekey and et.endpointkey=ep.endpointkey and ev.status=1 and ac.status=1
access.csv::::select distinct ENTITLEMENT_VALUE,et.ENTITLEMENTNAME,SOX_CRITICAL,SYS_CRITICAL,DESCRIPTION,UPDATEDATE as CREATIONDATE,SYSTEMNAME from entitlement_values ev ,entitlement_types et ,securitysystems ss where ev.entitlementtypekey=et.entitlementtypekey and et.systemkey = ss.systemkey and ev.status=1
ent2.csv::::select distinct CAST(ev1.entitlement_value AS Char(32)) as PARENT,cast(ev2.entitlement_value AS Char(32)) as CHILD from entitlements2 et ,entitlement_values ev1,entitlement_values ev2 where et.entitlement_value1key=ev1.entitlement_valuekey and et.entitlement_value2key=ev2.entitlement_valuekey and ev1.status=1 and ev2.status=1
