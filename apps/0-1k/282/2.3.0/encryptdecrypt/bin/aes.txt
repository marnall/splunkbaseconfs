>>> import aes
>>> text="Hello World. How you doing? It's great."
>>> password="mysecret"
>>> blocksize=256
>>> crypted = aes.encrypt( text, password, blocksize )
>>> crypted
'n5jJTbGxsbGS47pOXWsNYGFenJ70FC01PI4a/NH2Kf/ZKaFGk62VyN6xByxN5zw='
>>> aes.decrypt( crypted, password, blocksize )
"Hello World. How you doing? It's great."
>>> exit()
