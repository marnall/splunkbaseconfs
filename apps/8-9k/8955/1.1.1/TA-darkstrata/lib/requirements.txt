# Pinned for reproducible, lean packages.
#
# ucc-gen >= 6 requires splunktaucclib >= 8.2.0, so that is the floor here.
# solnlib >= 8 pulls in grpcio + opentelemetry + protobuf as hard dependencies,
# which bloat the package and trip AppInspect's check_hostnames_and_ips on the
# vendored binaries; solnlib 7.0.0 is the newest release without them and
# satisfies splunktaucclib's solnlib>=5 requirement. Pinning also keeps the
# bundled libraries stable across builds.
splunktaucclib==8.2.0
solnlib==7.0.0
